#Tartar Yoke
country_event = {
	id = InVH.90
	title = "InVH.EVTNAME90"
	desc = "InVH.EVTDESC90"
	picture = COURT_eventPicture

	fire_only_once = yes

	trigger = {
		tag = MOS
		is_year = 1300
		# is_subject_of = GOL 필요없는 것
		NOT = { is_year = 1460 }
	}

	mean_time_to_happen = {
		months = 1
	}

	option = {
		name = "InVH.EVTOPTA90"
		add_opinion = {
			who = GOL
			modifier = strong_distrust
		}
		GOL = {
			create_subject = {
				subject_type = independent_vassal
				subject = ROOT
			}
			hidden_effect = {
				add_opinion = {
					who = ROOT
					modifier = strong_distrust
				}
				ROOT = { add_liberty_desire = -30 }
			}
		}
	}
}
