namespace = InV
normal_or_historical_nations = yes

#Demand Diplomacy Rights
country_event = {
	id = InV.1
	title = "InV.EVTNAME1"
	desc = "InV.EVTDESC1"
	picture = COURT_eventPicture

	trigger = {
		ai = yes
		is_at_war = no
		is_subject = yes
		is_lesser_in_union = no
		liberty_desire = 50.0
		overlord = {
			NOT = {
				military_strength = {
					who = ROOT
					value = 1.2
				}
			}
		}
		OR = {
			is_subject_of_type = vassal
			is_subject_of_type = march
		}
	}

	mean_time_to_happen = {
		months = 60
	}

	option = {
		ai_chance = {
			factor = 2
			modifier = { factor = 2 overlord = { NOT = { military_strength = { who = ROOT value = 1.0 } } } }
		}
		ai_chance = {
			factor = 2
			modifier = { factor = 2 overlord = { NOT = { military_strength = { who = ROOT value = 0.8 } } } }
		}
		ai_chance = {
			factor = 2
			modifier = { factor = 2 overlord = { NOT = { military_strength = { who = ROOT value = 0.6 } } } }
		}
		name = "InV.EVTOPTA1"
		overlord = { country_event = { id = InV.2 days = 3 } }
	}
	option = {
		ai_chance = { factor = 1 }
		name = "InV.EVTOPTB1"
		add_liberty_desire = -10
	}
}



#Diplomacy Rights Demanded
country_event = {

	desc = "InV.EVTDESC2"

	}
}
